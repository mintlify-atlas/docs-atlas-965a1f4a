# Visualization Techniques

## Visualization Libraries

The project uses three main visualization libraries:

### Time Series Visualizations

#### Single Country Time Series

Visualize COVID-19 metrics over time for a specific country:

```python
import pandas as pd
import plotly.graph_objects as go

def plot_country_timeseries(df, country, metric, title=None):
    """
    Create an interactive time series plot for a country
    
    Args:
        df: DataFrame with COVID data
        country: Country name
        metric: Column name to plot
        title: Optional custom title
    """
    df_country = df[df['country'] == country].copy()
    
    fig = go.Figure()
    
    fig.add_trace(go.Scatter(
        x=df_country['date'],
        y=df_country[metric],
        mode='lines',
        name=metric,
        line=dict(color='steelblue', width=2),
        hovertemplate='<b>%{x}</b><br>%{y:.2f}<extra></extra>'
    ))
    
    fig.update_layout(
        title=title or f"{metric.replace('_', ' ').title()} - {country}",
        xaxis_title="Date",
        yaxis_title=metric.replace('_', ' ').title(),
        template="plotly_white",
        hovermode='x unified',
        height=500
    )
    
    return fig

# Example usage
fig = plot_country_timeseries(
    df,
    country='Spain',
    metric='new_deaths_smoothed_per_million',
    title='Daily COVID-19 Deaths in Spain (per million)'
)
fig.show()
```

#### Multi-Country Comparison

Compare multiple countries on the same plot:

```python
def plot_multi_country_comparison(df, countries, metric, title=None):
    """
    Compare metric across multiple countries
    """
    fig = go.Figure()
    
    colors = px.colors.qualitative.Set1
    
    for i, country in enumerate(countries):
        df_country = df[df['country'] == country]
        
        fig.add_trace(go.Scatter(
            x=df_country['date'],
            y=df_country[metric],
            mode='lines',
            name=country,
            line=dict(color=colors[i % len(colors)], width=2)
        ))
    
    fig.update_layout(
        title=title or f"{metric.replace('_', ' ').title()} - Country Comparison",
        xaxis_title="Date",
        yaxis_title=metric.replace('_', ' ').title(),
        template="plotly_white",
        hovermode='x unified',
        legend=dict(
            orientation="h",
            yanchor="bottom",
            y=1.02,
            xanchor="right",
            x=1
        ),
        height=500
    )
    
    return fig

# Example: Compare Spain, Italy, France
fig = plot_multi_country_comparison(
    df,
    countries=['Spain', 'Italy', 'France', 'Germany'],
    metric='new_cases_smoothed_per_million'
)
fig.show()
```

#### Dual-Axis Plots

Visualize two metrics with different scales:

```python
def plot_dual_axis(df, country, metric1, metric2):
    """
    Create dual-axis plot for two metrics
    """
    df_country = df[df['country'] == country].copy()
    
    fig = make_subplots(specs=[[{"secondary_y": True}]])
    
    # First metric (left axis)
    fig.add_trace(
        go.Scatter(
            x=df_country['date'],
            y=df_country[metric1],
            name=metric1.replace('_', ' ').title(),
            line=dict(color='steelblue', width=2)
        ),
        secondary_y=False
    )
    
    # Second metric (right axis)
    fig.add_trace(
        go.Scatter(
            x=df_country['date'],
            y=df_country[metric2],
            name=metric2.replace('_', ' ').title(),
            line=dict(color='orange', width=2)
        ),
        secondary_y=True
    )
    
    # Update axes
    fig.update_xaxes(title_text="Date")
    fig.update_yaxes(title_text=metric1.replace('_', ' ').title(), secondary_y=False)
    fig.update_yaxes(title_text=metric2.replace('_', ' ').title(), secondary_y=True)
    
    fig.update_layout(
        title=f"{country}: {metric1} vs {metric2}",
        template="plotly_white",
        hovermode='x unified',
        height=500
    )
    
    return fig

# Example: Cases vs Deaths
fig = plot_dual_axis(
    df,
    country='Spain',
    metric1='new_cases_smoothed_per_million',
    metric2='new_deaths_smoothed_per_million'
)
fig.show()
```

### Distribution Visualizations

#### Box Plots by Continent

```python
def plot_distribution_by_continent(df, metric):
    """
    Create box plot showing distribution by continent
    """
    # Get latest data point for each country
    df_latest = df.groupby(['continent', 'country']).last().reset_index()
    
    fig = px.box(
        df_latest,
        x='continent',
        y=metric,
        color='continent',
        points='all',  # Show all points
        hover_data=['country'],
        title=f"{metric.replace('_', ' ').title()} Distribution by Continent"
    )
    
    fig.update_traces(
        jitter=0.5,
        pointpos=0,
        marker=dict(size=6, opacity=0.8),
        line=dict(color='black')
    )
    
    fig.update_layout(
        showlegend=False,
        template="plotly_white",
        height=600
    )
    
    return fig

# Example usage
fig = plot_distribution_by_continent(df, 'total_deaths_per_million')
fig.show()
```

#### Violin Plots

```python
import seaborn as sns
import matplotlib.pyplot as plt

def plot_violin_comparison(df, metric, groupby='continent'):
    """
    Create violin plot with Seaborn
    """
    df_latest = df.groupby(['continent', 'country']).last().reset_index()
    
    plt.figure(figsize=(12, 6))
    sns.violinplot(
        data=df_latest,
        x=groupby,
        y=metric,
        palette='Set2',
        inner='box'
    )
    
    plt.title(f"{metric.replace('_', ' ').title()} Distribution by {groupby.title()}")
    plt.xlabel(groupby.title())
    plt.ylabel(metric.replace('_', ' ').title())
    plt.xticks(rotation=45)
    plt.tight_layout()
    plt.show()

# Example
plot_violin_comparison(df, 'vaccination_ratio', groupby='continent')
```

### Geographic Visualizations

#### Choropleth Map

Create interactive world map:

```python
def plot_choropleth_map(df, metric, date=None):
    """
    Create choropleth map for a specific metric
    """
    if date:
        df_map = df[df['date'] == date].copy()
    else:
        df_map = df.groupby('country').last().reset_index()
    
    fig = px.choropleth(
        df_map,
        locations='code',
        color=metric,
        hover_name='country',
        hover_data=[metric],
        color_continuous_scale='Reds',
        title=f"{metric.replace('_', ' ').title()} by Country"
    )
    
    fig.update_layout(
        template="plotly_white",
        height=600,
        geo=dict(
            showframe=False,
            showcoastlines=True,
            projection_type='natural earth'
        )
    )
    
    return fig

# Example: Total deaths per million
fig = plot_choropleth_map(df, 'total_deaths_per_million')
fig.show()
```

#### Animated Choropleth

Show evolution over time:

```python
def plot_animated_choropleth(df, metric, frame_step='M'):
    """
    Create animated choropleth map
    
    Args:
        df: DataFrame
        metric: Metric to visualize
        frame_step: 'D' (daily), 'W' (weekly), 'M' (monthly)
    """
    # Resample to reduce frame count
    df_anim = df.set_index('date').groupby('country').resample(frame_step).last().reset_index()
    
    fig = px.choropleth(
        df_anim,
        locations='code',
        color=metric,
        hover_name='country',
        animation_frame='date',
        color_continuous_scale='Reds',
        title=f"{metric.replace('_', ' ').title()} Over Time",
        range_color=[0, df_anim[metric].quantile(0.95)]  # Cap at 95th percentile
    )
    
    fig.update_layout(
        template="plotly_white",
        height=600,
        geo=dict(
            showframe=False,
            showcoastlines=True
        )
    )
    
    return fig

# Example: Animated total cases
fig = plot_animated_choropleth(df, 'total_cases_per_million', frame_step='M')
fig.show()
```

def plot_correlation_matrix(df, features):
    """
    Create correlation heatmap
    """
    corr = df[features].corr()
    
    plt.figure(figsize=(10, 8))
    sns.heatmap(
        corr,
        annot=True,
        fmt='.2f',
        cmap='coolwarm',
        center=0,
        square=True,
        linewidths=1,
        cbar_kws={"shrink": 0.8}
    )
    plt.title('Correlation Matrix', fontsize=14, pad=20)
    plt.tight_layout()
    plt.show()

def create_dashboard(df, country):
    """
    Create a comprehensive dashboard for a country
    """
    df_country = df[df['country'] == country].copy()
    
    fig = make_subplots(
        rows=2, cols=2,
        subplot_titles=(
            'Daily Cases',
            'Daily Deaths',
            'Vaccination Progress',
            'Stringency Index'
        )
    )
    
    # Cases
    fig.add_trace(
        go.Scatter(x=df_country['date'], y=df_country['new_cases_smoothed_per_million'],
                   name='Cases', line=dict(color='blue')),
        row=1, col=1
    )
    
    # Deaths
    fig.add_trace(
        go.Scatter(x=df_country['date'], y=df_country['new_deaths_smoothed_per_million'],
                   name='Deaths', line=dict(color='red')),
        row=1, col=2
    )
    
    # Vaccination
    fig.add_trace(
        go.Scatter(x=df_country['date'], y=df_country['vaccination_ratio'],
                   name='Vaccination', line=dict(color='green')),
        row=2, col=1
    )
    
    # Stringency
    fig.add_trace(
        go.Scatter(x=df_country['date'], y=df_country['stringency_index'],
                   name='Stringency', line=dict(color='orange')),
        row=2, col=2
    )
    
    fig.update_layout(
        title=f"COVID-19 Dashboard: {country}",
        template="plotly_white",
        height=800,
        showlegend=False
    )
    
    return fig

## Styling Best Practices

<Tip>
**Visualization tips:**

1. **Color choice**: Use colorblind-friendly palettes (e.g., `Set2`, `viridis`)
2. **Axis labels**: Always include clear, descriptive labels with units
3. **Titles**: Make titles informative and specific
4. **Legends**: Position legends to avoid obscuring data
5. **Annotations**: Add annotations for important events (e.g., policy changes)
6. **Consistency**: Use the same color scheme across related visualizations
</Tip>

## Performance Optimization

<Warning>
For large datasets:
- **Downsample** time series data when creating visualizations
- **Aggregate** data before plotting (e.g., weekly instead of daily)
- **Limit points** in scatter plots using sampling
- **Use WebGL** for Plotly plots with many points: `fig.update_traces(mode='markers', marker=dict(size=3))`
</Warning>
